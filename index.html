Assessment

Fix the ShowUp Room Feed

Users in the same room cannot see each other's posts in real time and posts do not persist after a page refresh.

Your tasks
01Diagnose what you believe is causing the issue.
02Propose a fix, either in code or with a clear written explanation.
03Optional: implement the fix if you'd like to demonstrate your approach.
Reply to the email with your response within 48 hours of receiving it.
Code — ShowUp.jsx
import { useMemo, useState } from 'react'
import {
  ArrowLeft,
  Check,
  ChevronRight,
  Image as ImageIcon,
  MessageCircle,
  Send,
  ThumbsUp,
  Users,
} from 'lucide-react'

const ROOM_DEFINITIONS = [
  { id: 'personal-growth', name: 'Personal Growth', pillar: 'Personal Growth', maxSpots: 12 },
  { id: 'health-fitness', name: 'Health & Fitness', pillar: 'Health & Fitness', maxSpots: 12 },
  { id: 'career-business', name: 'Career & Business', pillar: 'Career & Business', maxSpots: 12 },
  { id: 'wealth', name: 'Wealth', pillar: 'Wealth', maxSpots: 12 },
  { id: 'relationships', name: 'Relationships', pillar: 'Relationships', maxSpots: 12 },
  { id: 'inner-life', name: 'Inner Life', pillar: 'Inner Life', maxSpots: 12 },
]

const seedMembers = [
  { id: 'u1', name: 'Favour Addy', initials: 'FA', role: 'Room Leader', taskDone: true, streak: 8 },
  { id: 'u2', name: 'Big Baby Andy', initials: 'BB', role: 'Comeback', taskDone: false, streak: 3 },
  { id: 'u3', name: 'You', initials: 'YO', role: '', taskDone: false, streak: 1 },
]

const seedFeed = [
  {
    id: 'p1',
    author: 'Sage',
    time: 'May 21 at 01:26 AM',
    text: 'The reason most people do not finish things is unclear next steps. What is your next step today?',
    image: '',
    likes: 0,
    comments: [],
    system: true,
  },
]

function buildInitials(name) {
  return String(name || 'User')
    .trim()
    .split(/\s+/)
    .slice(0, 2)
    .map(part => part[0]?.toUpperCase() || '')
    .join('') || 'U'
}

function formatTime(value) {
  if (!value) return ''
  return value
}

function getEnergyLabel(doneCount) {
  if (doneCount === 0) return 'Quiet'
  if (doneCount <= 2) return 'Warming Up'
  if (doneCount <= 5) return 'Focused'
  if (doneCount <= 8) return 'On Fire'
  if (doneCount <= 11) return 'Locked In'
  return 'Full Send'
}

export default function ShowUpRoomHandoff() {
  const [selectedRoom, setSelectedRoom] = useState(null)
  const [activeTab, setActiveTab] = useState('live')
  const [checkedIn, setCheckedIn] = useState(false)
  const [taskDone, setTaskDone] = useState(false)
  const [checkInTime, setCheckInTime] = useState('')
  const [toast, setToast] = useState('')
  const [progressToast, setProgressToast] = useState('')
  const [rooms] = useState(ROOM_DEFINITIONS.map(room => ({ ...room, currentMembers: 0 })))
  const [members, setMembers] = useState(seedMembers)
  const [feedPosts, setFeedPosts] = useState(seedFeed)
  const [postDraft, setPostDraft] = useState('')
  const [postImage, setPostImage] = useState('')
  const [commentSheetPostId, setCommentSheetPostId] = useState('')
  const [commentDraft, setCommentDraft] = useState('')
  const [replyDraft, setReplyDraft] = useState('')

  const room = useMemo(
    () => ROOM_DEFINITIONS.find(item => item.name === selectedRoom) || null,
    [selectedRoom]
  )

  const doneCount = members.filter(member => member.taskDone).length
  const energy = getEnergyLabel(doneCount)

  function handleJoinRoom(roomName) {
    setSelectedRoom(roomName)
    setActiveTab('live')
    setCheckedIn(false)
    setTaskDone(false)
    setCheckInTime('')
    setToast('')
  }

  function handleCheckIn() {
    const now = new Date().toLocaleTimeString('en-US', {
      hour: '2-digit',
      minute: '2-digit',
    })
    setCheckedIn(true)
    setCheckInTime(now)
  }

  function handleMarkDone() {
    if (!checkedIn || taskDone) return
    setTaskDone(true)
    setProgressToast('Nice work. Post a progress photo.')
    setTimeout(() => setActiveTab('feed'), 1500)
    setTimeout(() => setProgressToast(''), 3000)
  }

  function handleCreatePost() {
    if (!postDraft.trim() && !postImage) return
    setFeedPosts(current => [
      {
        id: `post-${Date.now()}`,
        author: 'You',
        time: 'Just now',
        text: postDraft.trim(),
        image: postImage,
        likes: 0,
        comments: [],
        system: false,
      },
      ...current,
    ])
    setPostDraft('')
    setPostImage('')
  }

  function handleAddComment(postId) {
    if (!commentDraft.trim()) return
    setFeedPosts(current =>
      current.map(post => {
        if (post.id !== postId) return post
        return {
          ...post,
          comments: [
            ...post.comments,
            {
              id: `comment-${Date.now()}`,
              author: 'You',
              text: commentDraft.trim(),
              replies: replyDraft.trim()
                ? [{ id: `reply-${Date.now()}`, author: 'You', text: replyDraft.trim() }]
                : [],
            },
          ],
        }
      })
    )
    setCommentDraft('')
    setReplyDraft('')
  }

  // ... rest of component
}
